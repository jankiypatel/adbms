DECLARE
   CNT NUMBER;
   ED NUMBER;
   X  NUMBER;
   DA NUMBER;
   HRA NUMBER;
   NET_SAL NUMBER;

BEGIN
X:=:ENTERNO;
   

BEGIN

SELECT COUNT(*) INTO "CNT" FROM EMPS WHERE EMP_NO=X;
IF CNT >0 THEN

  SELECT SALARY INTO "ED" FROM EMPS WHERE EMP_NO=X; 
  HRA:=(ED * 10)/100;
  DA:=(ED * 60)/100;
  NET_SAL:=ED+DA+HRA;

  DBMS_OUTPUT.PUT_LINE('NETSAL=' || NET_SAL );

ELSE
  DBMS_OUTPUT.PUT_LINE('INVALID SAL');

END IF;

END; 

END;
